重启命令
reboot shutdown -r for now
强制删除目录
rm -rf directory
查看当前目录下的文件信息
ls -l
查看当前目录下文件个数
find ./ | wc -l
切换到上一个工作目录
cd -
切换到root目录
cd ~
显示当前路径
pwd
查找当前目录下以sql开头的文件或者目录
find ./ -name "sql*"
查看文件内容 和 查看文件内容并且显示行号
cat filename cat -n filename
查看文件内容的前10行 和 查看文件内容的后5行
head -10 filename tail -5 filename
查看两个文件间的差别
diff filename1 filename2
查看磁盘空间利用大小
df -h
查看当前目录所占空间大小
du -sh
查看当前目录下所有子文件夹排序后的大小
for i in `ls`; do du -sh $i; done | sort du -sh `ls` | sort

tar -cvf etc.tar /etc

gzip demo.txt
